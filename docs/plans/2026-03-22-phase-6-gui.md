# Phase 6: GUI — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a native desktop GUI for Surge using egui/eframe — dashboard, kanban board, execution monitor, agent hub, diff viewer, and terminal panel.

**Architecture:** New `surge-ui` crate with `eframe` for windowing and `egui` for immediate-mode UI. App struct holds shared state via `Arc<Mutex<AppState>>`. Panels are rendered by trait objects implementing a `Panel` trait. Tokio runtime bridged via channels for async operations (agent connections, pipeline events). Dark theme matching Surge brand.

**Tech Stack:** Rust 2024, eframe 0.31, egui 0.31, tokio (runtime bridge), surge-core/spec/git/acp/orchestrator

//! Surge dark theme for egui.

use eframe::egui::{self, Color32, Rounding, Stroke, Style, Visuals};

/// Surge brand colors.
pub struct SurgeColors;

impl SurgeColors {
    pub const BG_DARK: Color32 = Color32::from_rgb(17, 17, 27);
    pub const BG_PANEL: Color32 = Color32::from_rgb(24, 24, 37);
    pub const BG_CARD: Color32 = Color32::from_rgb(30, 30, 46);
    pub const BORDER: Color32 = Color32::from_rgb(49, 50, 68);
    pub const TEXT: Color32 = Color32::from_rgb(205, 214, 244);
    pub const TEXT_DIM: Color32 = Color32::from_rgb(147, 153, 178);
    pub const ACCENT: Color32 = Color32::from_rgb(137, 180, 250);    // blue
    pub const SUCCESS: Color32 = Color32::from_rgb(166, 227, 161);   // green
    pub const WARNING: Color32 = Color32::from_rgb(249, 226, 175);   // yellow
    pub const ERROR: Color32 = Color32::from_rgb(243, 139, 168);     // red
    pub const PURPLE: Color32 = Color32::from_rgb(203, 166, 247);    // agent badge
}

/// Apply the Surge dark theme to egui context.
pub fn apply_theme(ctx: &egui::Context) {
    let mut style = Style::default();

    let mut visuals = Visuals::dark();
    visuals.panel_fill = SurgeColors::BG_PANEL;
    visuals.window_fill = SurgeColors::BG_CARD;
    visuals.extreme_bg_color = SurgeColors::BG_DARK;
    visuals.faint_bg_color = SurgeColors::BG_CARD;
    visuals.window_rounding = Rounding::same(8.0);
    visuals.window_stroke = Stroke::new(1.0, SurgeColors::BORDER);
    visuals.override_text_color = Some(SurgeColors::TEXT);

    style.visuals = visuals;
    ctx.set_style(style);
}

/// Panel navigation.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
pub enum ActivePanel {
    #[default]
    Dashboard,
    Kanban,
    Execution,
    AgentHub,
    DiffViewer,
    Terminal,
}

impl ActivePanel {
    pub fn label(&self) -> &'static str {
        match self {
            Self::Dashboard => "Dashboard",
            Self::Kanban => "Kanban Board",
            Self::Execution => "Live Execution",
            Self::AgentHub => "Agent Hub",
            Self::DiffViewer => "Diff Viewer",
            Self::Terminal => "Terminal",
        }
    }

    pub fn icon(&self) -> &'static str {
        match self {
            Self::Dashboard => "\u{1F3E0}",
            Self::Kanban => "\u{2B1C}",
            Self::Execution => "\u{1F3C3}",
            Self::AgentHub => "\u{1F50C}",
            Self::DiffViewer => "\u{1F4C4}",
            Self::Terminal => "\u{1F5A5}",
        }
    }

    pub fn all() -> &'static [ActivePanel] {
        &[
            Self::Dashboard,
            Self::Kanban,
            Self::Execution,
            Self::AgentHub,
            Self::DiffViewer,
            Self::Terminal,
        ]
    }
}

//! Dashboard panel — project overview.

use eframe::egui;
use crate::theme::SurgeColors;

pub fn show(ui: &mut egui::Ui) {
    ui.heading("Dashboard");
    ui.add_space(16.0);

    ui.horizontal(|ui| {
        stat_card(ui, "Tasks", "0", SurgeColors::ACCENT);
        stat_card(ui, "Active", "0", SurgeColors::SUCCESS);
        stat_card(ui, "Failed", "0", SurgeColors::ERROR);
        stat_card(ui, "Agents", "0", SurgeColors::PURPLE);
    });

    ui.add_space(24.0);
    ui.label("Recent Activity");
    ui.separator();
    ui.colored_label(SurgeColors::TEXT_DIM, "No recent activity. Create a spec to get started.");
}

fn stat_card(ui: &mut egui::Ui, label: &str, value: &str, color: egui::Color32) {
    egui::Frame::new()
        .fill(SurgeColors::BG_CARD)
        .corner_radius(8.0)
        .inner_margin(16.0)
        .stroke(egui::Stroke::new(1.0, SurgeColors::BORDER))
        .show(ui, |ui| {
            ui.vertical(|ui| {
                ui.colored_label(color, value);
                ui.colored_label(SurgeColors::TEXT_DIM, label);
            });
        });
}

**Step 4: Write app.rs**

```rust
//! Main application struct.

use eframe::egui;
use crate::panels::ActivePanel;
use crate::theme;

pub struct SurgeApp {
    active_panel: ActivePanel,
}

impl Default for SurgeApp {
    fn default() -> Self {
        Self {
            active_panel: ActivePanel::Dashboard,
        }
    }
}

impl SurgeApp {
    pub fn new(cc: &eframe::CreationContext<'_>) -> Self {
        theme::apply_theme(&cc.egui_ctx);
        Self::default()
    }
}

impl eframe::App for SurgeApp {
    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
        // Sidebar
        egui::SidePanel::left("sidebar")
            .resizable(false)
            .default_width(200.0)
            .show(ctx, |ui| {
                ui.add_space(8.0);
                ui.heading("⚡ Surge");
                ui.add_space(16.0);
                ui.separator();
                ui.add_space(8.0);

                for panel in ActivePanel::all() {
                    let selected = self.active_panel == *panel;
                    let label = format!("{} {}", panel.icon(), panel.label());
                    if ui.selectable_label(selected, label).clicked() {
                        self.active_panel = *panel;
                    }
                }
            });

        // Main content
        egui::CentralPanel::default().show(ctx, |ui| {
            match self.active_panel {
                ActivePanel::Dashboard => crate::panels::dashboard::show(ui),
                ActivePanel::Kanban => crate::panels::kanban::show(ui),
                ActivePanel::Execution => crate::panels::execution::show(ui),
                ActivePanel::AgentHub => crate::panels::agent_hub::show(ui),
                ActivePanel::DiffViewer => crate::panels::diff_viewer::show(ui),
                ActivePanel::Terminal => crate::panels::terminal::show(ui),
            }
        });
    }
}
```

**Step 5: Write main.rs**

```rust
//! Surge GUI application.

mod app;
mod panels;
mod theme;

fn main() -> eframe::Result<()> {
    tracing_subscriber::fmt()
        .with_env_filter(
            tracing_subscriber::EnvFilter::try_from_default_env()
                .unwrap_or_else(|_| "surge_ui=info".into()),
        )
        .init();

    let options = eframe::NativeOptions {
        viewport: eframe::egui::ViewportBuilder::default()
            .with_inner_size([1280.0, 800.0])
            .with_min_inner_size([800.0, 600.0])
            .with_title("Surge — Agent Orchestrator"),
        ..Default::default()
    };

    eframe::run_native(
        "Surge",
        options,
        Box::new(|cc| Ok(Box::new(app::SurgeApp::new(cc)))),
    )
}
```
